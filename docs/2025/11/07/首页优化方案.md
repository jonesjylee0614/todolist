export function applyFilter(tasks: TaskDTO[], filter: string): TaskDTO[] {
  switch (filter) {
    case 'all':
      return tasks;
    case 'with-deadline':
      return tasks.filter(t => !!t.deadline);
    case 'overdue':
      return tasks.filter(t => t.deadline && isOverdue(t.deadline));
    case 'today':
      return tasks.filter(t => t.deadline && isToday(new Date(t.deadline)));
    case 'this-week':
      return tasks.filter(t => t.deadline && isThisWeek(new Date(t.deadline)));
    case 'no-deadline':
      return tasks.filter(t => !t.deadline);
    default:
      return tasks;
  }
}

export function applySort(tasks: TaskDTO[], sort: string): TaskDTO[] {
  const sorted = [...tasks];
  switch (sort) {
    case 'deadline':
      return sorted.sort(compareByDeadline);
    case 'created':

      );
    case 'title':
      return sorted.sort((a, b) => a.title.localeCompare(b.title));

      return sorted;
